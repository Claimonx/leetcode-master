class Solution:
    def strStr(self, haystack: str, needle: str) -> int:
        # 生成next数组（其中i是高歌猛进的，不会回退，j进行回退操作）
        # 1、初始化
        # 2、前缀后缀相同
        # 3、前缀后缀不相同
        # 4、next更行
        pattern_len = len(needle)
        j = 0  # 前缀末尾
        next = [0]
        i = 1  # 后缀末尾
        while i < pattern_len:
            # 前缀后缀不相同,j(前缀)回退操作
            while j > 0 and needle[i] != needle[j]:
                j = next[j - 1]
            # 前后后缀相同
            if needle[i] == needle[j]:
                j += 1
            # next 数组更新,并且i进位
            # if needle[j] == needle[next[j-1]]:  #  nextval 操作
            #     next.append(next[j-1])
            # else:
            next.append(j)
            i += 1

        print(next)

        # KMP查询匹配操作
        i = 0  # 主串索引位置
        j = 0  # 模式匹配字串位置
        main_len = len(haystack)
        while i < main_len and j < pattern_len:
            # 回退的情况：不相等，且不是第一位
            if haystack[i] != needle[j] and j != 0:
                j = next[j - 1]
            elif haystack[i] != needle[j] and j == 0:
                i += 1
            else:
                i += 1
                j += 1

        # 全部匹配完
        if j == pattern_len:
            return i - pattern_len
        else:
            return -1
